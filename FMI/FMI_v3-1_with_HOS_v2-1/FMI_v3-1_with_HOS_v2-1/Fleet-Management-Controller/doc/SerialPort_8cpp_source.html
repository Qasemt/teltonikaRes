178 void SerialPort::pumpRx()
179 {
180  /*----------------------------------------------------------
181  Local Variables
182  ----------------------------------------------------------*/
183  uint8 readBuffer[ 256 ];
184  int readSize;
185  DWORD error;
186  COMSTAT status;
187 
188  //check for error
189  ClearCommError( mComPortHandle, &error, &status );
190  if( error != 0 )
191  {
192  PurgeComm( mComPortHandle, PURGE_RXABORT | PURGE_RXCLEAR );
193  return;
194  }
195 
196  //check read size...if zero return because nothing to do
197  readSize = minval( status.cbInQue, sizeof( readBuffer ) );
198  if( readSize == 0 )
199  return;
200 
201  if( ReadFile( mComPortHandle, readBuffer, readSize, (LPDWORD)&readSize, NULL ) == 0 )
202  {
203  ClearCommError( mComPortHandle, &error, &status );
204  if( ( error & CE_OVERRUN ) ||
205  ( error & CE_RXOVER ) )
206  {
207  PurgeComm( mComPortHandle, PURGE_RXABORT | PURGE_RXCLEAR );
208  }
209  return;
210  }
211  if( readSize == 0 )
212  return;
213 
214  // Push the received bytes up to the link layer
215  if( mLinkLayer )
216  {
217  mLinkLayer->rx( readBuffer, readSize );
218  }
219 }

262 bool SerialPort::getPortListFromRegistry
263  (
264  std::list<CString> &aList
265  )
266 {
267  HKEY hKey; // handle for the key being queried
268  DWORD numValues; // number of values for key
269  DWORD maxValueLen; // longest value name
270  DWORD maxDataLen; // longest value data
271  DWORD dataType; // out: code indicating type of data
272  BYTE *dataBuf; // out: data buffer
273  DWORD dataLen; // out: number of bytes
274  DWORD i, retCode;
275 
276  TCHAR *valueBuf;
277  DWORD valueLen;
278 
279  bool retval = false;
280 
281  if( RegOpenKeyEx
282  (
283  HKEY_LOCAL_MACHINE,
284  TEXT("HARDWARE\\DEVICEMAP\\SERIALCOMM"),
285  0,
286  KEY_READ,
287  &hKey
288  ) == ERROR_SUCCESS )
289  {
290  // Get the class name and the value count.
291  retCode = RegQueryInfoKey
292  (
293  hKey, // key handle
294  NULL, NULL, // buffer and size of class name, don't care
295  NULL, // reserved
296  NULL, NULL, // number and length of subkeys, don't care
297  NULL, // longest class string, don't care
298  &numValues, // number of values for this key
299  &maxValueLen, // longest value name
300  &maxDataLen, // longest value data
301  NULL, NULL // security descriptor, last write time, don't care
302  );
303 
304  // Enumerate the values and data under this key.
305  if( numValues )
306  {
307  valueBuf = new TCHAR[maxValueLen + 1];
308  dataBuf = new BYTE[maxDataLen + 1];
309  retval = true;
310 
311  for( i = 0, retCode = ERROR_SUCCESS; i < numValues; i++ )
312  {
313  valueLen = maxValueLen + 1;
314  dataLen = maxDataLen + 1;
315  valueBuf[0] = '\0';
316  retCode = RegEnumValue
317  (
318  hKey,
319  i,
320  valueBuf,
321  &valueLen,
322  NULL,
323  &dataType,
324  dataBuf,
325  &dataLen
326  );
327  if( retCode == ERROR_SUCCESS )
328  {
329  if( dataType == REG_SZ )
330  aList.push_back( CString( (TCHAR*)dataBuf ) );
331  }
332  else
333  retval = false;
334 
335  } // end of for loop
336 
337  delete[] valueBuf;
338  delete[] dataBuf;
339  } // end of if( numValues )
340 
341  } // end of if( RegOpenKey )
342 
343  RegCloseKey( hKey );
344  return retval;
345 }
