216  CString path;
217  for ( int i=0; i<folders.GetCount(); i++ )
218  {
219  path += folders[i] + _T( "\\" );
220  if ( ERROR_PATH_NOT_FOUND == CreateDirectory( path, NULL) )

222  MessageBox( _T( "Failed to create repository folder " ) + path, _T( "Form Submitted" ), MB_OK | MB_ICONERROR );

442 void CCustomFormsDlg::OnBeforeNavigate2(LPDISPATCH, VARIANT* URL, VARIANT*, VARIANT*, VARIANT*, VARIANT*, BOOL* Cancel)
443  {
444  if ( URL->vt != VT_BSTR )
445  {
446  return;
447  }
448 
449  CString urlStr = URL->bstrVal;
450  // manually transform form templates and submitted forms since the XSL templates do not know where the
451  // supporting resources are located
452  if ( urlStr.Find( _T( "template:" ) ) == 0)
453  {
454  // take over
455  *Cancel = TRUE;
456 
457  // override relative path param defaults in the template (which were set for debugging purposes)
458  CMapStringToString params;
459  // resources are relative to the temp file
460  params.SetAt( _T( "resources" ), _T( "" ) );
461  CString src = CWebBrowser2::URLDecode( urlStr.Right( urlStr.GetLength() - (int) strlen( "template:" ) ) );
462  transform( src, m_templateTransform, params );
463  }
464  else if ( urlStr.Find( _T( "submitted:") ) == 0 )
465  {
466  // take over
467  *Cancel = TRUE;
468 
469  CString submittedFile = CWebBrowser2::URLDecode( urlStr.Right( urlStr.GetLength() - (int) strlen( "submitted:" ) ) );
470 
471  // override relative path param defaults in the template (which were set for debugging purposes)
472  CMapStringToString params;
473  // resources are relative to the temp file
474  params.SetAt( _T( "resources" ), _T( "" ) );
475  // the form template file is in the same folder as the submitted form file
476  CString formPath = submittedFile.Left( submittedFile.ReverseFind( '/' ) + 1 );
477  params.SetAt( _T( "formPath" ), formPath );
478  transform( submittedFile, m_submittedTransform, params );
479  }
480  else if ( urlStr.Find( _T( "app:") ) == 0 )
481  {
482  // take over
483  *Cancel = TRUE;
484 
485  CString command = urlStr.Right( urlStr.GetLength() - (int) strlen( "app:" ) );
486  CString procedure = command.Left( command.Find( '(' ) );
487  CString args = command.Mid( procedure.GetLength() + 1, command.ReverseFind( ')') - ( procedure.GetLength() + 1 ) );
488 
489  TRACE2( "app command '%s' executed with args '%s'\r\n", procedure, args );
490 
491  CStringArray arguments;
492  //split arguments separated by comma
493  int curPos = 0;
494  CString token = args.Tokenize( _T( "," ), curPos );
495  while ( token != _T( "" ) )
496  {
497  if ( 0 == token.Find( '"' ) )
498  {
499  if ( token.GetLength()-1 == token.ReverseFind( '"' ) )
500  {
501  token = token.Mid( 1, token.GetLength() - 2 );
502  token = CWebBrowser2::URLDecode( token );
503  }
504  else {
505  token += _T( "," );
506  token += args.Tokenize( _T( "," ), curPos );
507  continue;
508  }
509  }
510  arguments.Add( token );
511  token = args.Tokenize( _T( "," ), curPos );
512  }
513 
514  if ( procedure == _T( "importForm" ) )
515  {
516  importForm();
517  }
518  else if ( procedure == _T( "deleteCustom" ) )
519  {
520  deleteFromDevice( _ttoi( arguments[0] ) );
521  }
522  else if ( procedure == _T( "changePosition" ) )
523  {
524  moveOnDevice( _ttoi( arguments[0] ), (uint8) _ttoi( arguments[1] ) );
525  }
526  else if ( procedure == _T( "requestPosition" ) )
527  {
528  requestPosition( _ttoi( arguments[0] ) );
529  }
530  else if ( procedure == _T( "sendForm" ) )
531  {
532  CString formFile = arguments[0];
533  formFile.Replace( '/', '\\');
534  sendToDevice( formFile.Right( formFile.GetLength() - (int) strlen( "file:///" ) ) );
535  }
536  else if ( procedure == _T( "removeForm" ) )
537  {
538  CString formFile = arguments[0];
539  formFile.Replace( '/', '\\');
540  if ( DeleteFile( formFile.Right( formFile.GetLength() - (int) strlen( "file:///" ) ) ) )
541  {
542  refreshCatalog();
543  }
544  else {
545  CString message;
546  message.Format( _T( "Unable to remove form. Error code: %d" ), GetLastError() );
547  MessageBox( message, _T( "Remove form" ), MB_OK | MB_ICONERROR );
548  }
549  }
550  }

553 BOOL CCustomFormsDlg::importForm()
554  {

556  // opening a file in another directory changes the current
557  // directory, which will cause problems because the log and data
558  // files are opened relative to the current directory. So, get the
559  // directory now so that it can be restored when the user is done
560  // picking a file.

686 BOOL CCustomFormsDlg::sendToDevice( LPCTSTR filename )
687  {
688  CFileStatus fileStatus;
689  if ( !CFile::GetStatus( filename, fileStatus ) )
690  {
691  CString message;
692  message.Format( _T( "Form file %s does not exist." ), filename );
693  MessageBox( message, _T( "Send Form Template to Device" ), MB_OK | MB_ICONERROR );
694  return FALSE;
695  }
696 
697  // validate on load
698 
699  // specify the schema for the form template validator since it isn't in same folder as loaded files (and cache it)
700  MSXML2::IXMLDOMSchemaCollection2Ptr templateSchemas( MSXML2::CLSID_XMLSchemaCache60 );
701  templateSchemas->add( _bstr_t( "" ), _bstr_t( "custom-forms\\form_template.xsd" ) );
702 
703  MSXML2::IXMLDOMDocument2Ptr templateDoc( MSXML2::CLSID_DOMDocument60 );
704  templateDoc->schemas = templateSchemas.GetInterfacePtr();;
705 
706  templateDoc->async = VARIANT_FALSE;
707  templateDoc->validateOnParse = VARIANT_TRUE;
708  templateDoc->resolveExternals = VARIANT_TRUE;
709 
710  VARIANT_BOOL valid = templateDoc->load( _variant_t( filename ) );
711  if ( VARIANT_TRUE != valid )
712  {
713  if (IDNO == MessageBox(
714  _T( "Form validation failed.\nDo you want to send the file anyway?" ),
715  _T( "Send Form Template to Device" ),
716  MB_YESNO | MB_ICONWARNING ) )
717  {
718  return FALSE;
719  }
720  // reload without validation
721  templateDoc->validateOnParse = VARIANT_FALSE;
722  templateDoc->load( _variant_t( filename ) );
723  }
724 
725  MSXML2::IXMLDOMNodePtr formVersion = templateDoc->selectSingleNode( _bstr_t( "//form/@version" ) );
726  if ( formVersion == NULL)
727  {
728  MessageBox(
729  _T( "Form template does not specify a version attribute, which is necessary for file transport." ),
730  _T( "Send Form Template to Device" ),
731  MB_OK | MB_ICONERROR );
732  return FALSE;
733  }
734 
735  char file[MAX_PATH+1];
736  char versionString[35];
737  uint8 version[16];
738  uint8 versionLength;
739 
740  memset( version, 0, sizeof( version ) );
741  WideCharToMultiByte( mCom.mClientCodepage, 0, formVersion->Gettext(), -1, versionString, 34, NULL, NULL );
742  versionString[34] = '\0';
743  versionLength = (uint8)minval( 16, strlen( versionString ) );
744  memmove( version, versionString, versionLength );
745 
746  WideCharToMultiByte( CP_ACP, 0, filename, -1, file, MAX_PATH+1, NULL, NULL );
747  file[MAX_PATH] = '\0';
748 
749  mCom.sendFile( file, versionLength, version, (uint8) FMI_FILE_TYPE_CUSTOM_FORMS, TRUE );
750 
751  CFileTransferProgressDlg sending_dlg( this, mCom );
752  sending_dlg.DoModal();
753 
754  return TRUE;
755  }
