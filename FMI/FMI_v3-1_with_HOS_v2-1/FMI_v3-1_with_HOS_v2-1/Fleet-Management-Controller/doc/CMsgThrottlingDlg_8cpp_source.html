157 void CMsgThrottlingDlg::OnBnClickedUpdate()
158 {
159  int i;
160 
161  UpdateData( TRUE );
162 
163  // Check to see which protocols actually changed
164  for( i = 0; i < NUM_THROTTLED_PROTOCOLS; ++i )
165  {
166  if( *( mProtocolStateLookup[i] ) != mOriginalValues[i] )
167  {
168  mCom.sendMessageThrottlingUpdate
169  (
170  mPacketIdLookup[i],
171  ( *mProtocolStateLookup[i] ) ? (uint16) MESSAGE_THROTTLE_STATE_DISABLE : (uint16) MESSAGE_THROTTLE_STATE_ENABLE
172  );
173  }
174  mOriginalValues[i] = *mProtocolStateLookup[i];
175  }
176 
177  OnOK();
178 } /* OnBnClickedUpdate() */

209 afx_msg LRESULT CMsgThrottlingDlg::OnThrottleQueryResponse
210  (
211  WPARAM aResponseCount,
212  LPARAM aResponseList
213  )
214 {
215  message_throttling_data_type * list = (message_throttling_data_type *)aResponseList;
216  uint32 listCount = (uint32)aResponseCount;
217 
218  UpdateData( TRUE );

222  {
223  if( mPacketIdLookup[localIdx] == list[listIdx].packet_id )
