128 bool GarminLinkLayer::getPacketFromFifo()
129 {
130  /*----------------------------------------------------------
131  Local Variables
132  ----------------------------------------------------------*/
133  uint32 fifoSize;
134  header_type header;
135  footer_type footer;
136  uint32 prevHead;
137  uint32 payloadStart;
138  uint32 payloadEnd;
139  uint8 checksum;
140  uint8 *ptr;
141  GarminPacket *packet;
142 
143  //loop until Garmin back found or FIFO empty
144  while( mFifo.head != mFifo.tail )
145  {
146  prevHead = mFifo.head;
147  fifoSize = numBytesInFifo();
148  if( fifoSize < MIN_PACKET_SIZE ) return false;
149 
150  getDataFromFifo( (uint8 *)&header, SIZE_OF_HEADER );
151 
152  //find header
153  if( ( header.dle != ID_DLE_BYTE ) ||
154  ( header.id == ID_ETX_BYTE ) )
155  {
156  incrementHead( 1 );
157  continue;
158  }
159  //find footer
160  incrementHead( SIZE_OF_HEADER );
161  payloadStart = mFifo.head;
162  payloadEnd = payloadStart;
163  while( mFifo.head != mFifo.tail )
164  {
165  fifoSize = numBytesInFifo();
166  if( fifoSize < SIZE_OF_FOOTER )
167  {
168  break;
169  }
170 
171  getDataFromFifo( (uint8*)&footer, SIZE_OF_FOOTER );
172  if( ( footer.dle == ID_DLE_BYTE ) &&
173  ( footer.etx == ID_DLE_BYTE ) )
174  {
175  incrementHead( 2 );
176  }
177  else if( ( footer.dle == ID_DLE_BYTE ) &&
178  ( footer.etx == ID_ETX_BYTE ) )
179  {
180  payloadEnd = mFifo.head;
181  break;
182  }
183  else
184  incrementHead( 1 );
185  }
186  //if footer not found, entire packet not received yet
187  if( payloadStart == payloadEnd )
188  {
189  mFifo.head = prevHead;
190  return false;
191  }
192 
193  //complete packet exists in receive queue
194  packet = new GarminPacket;
195  mFifo.head = payloadStart;
196  packet->mPacketId = header.id;
197  packet->mPayloadSize = header.size;
198  if( packet->mPayloadSize == ID_DLE_BYTE )
199  {
200  incrementHead( 1 );
201  }
202  packet->mFrameSize = 0;
203  packet->mChecksum = ( header.id + header.size ) % 256;
204 
205  ptr = (uint8*)packet->mPayload;
206 
207  while( ( mFifo.head != payloadEnd )
208  && ( packet->mFrameSize < packet->mPayloadSize ) )
209  {
210  getDataFromFifo( (uint8*)&ptr[packet->mFrameSize], 2 );
211  packet->mChecksum = ( ptr[ packet->mFrameSize] + packet->mChecksum ) % 256;
212  incrementHead( 1 );
213  if( ( ptr[ packet->mFrameSize ] == ID_DLE_BYTE ) &&
214  ( ptr[ packet->mFrameSize + 1 ] == ID_DLE_BYTE ) )
215  incrementHead( 1 );
216  packet->mFrameSize++;
217  }
218 
219  getDataFromFifo( (uint8*)&checksum, 1 );
220  packet->mChecksum = ( checksum + packet->mChecksum ) % 256;
221  if( checksum == ID_DLE_BYTE )
222  {
223  incrementHead( 2 );
224  }
225  else
226  {
227  incrementHead( 1 );
228  }
229 
230  packet->mFrameSize = 0;
231  uint32 endOfFrame = (mFifo.head + SIZE_OF_FOOTER) % FIFO_SIZE;
232  while( prevHead != endOfFrame )
233  {
234  packet->mFrame[packet->mFrameSize++] = mFifo.buffer[prevHead];
235  //increment prevHead and wrap around to 0 if goes past FIFO boundaries
236  prevHead = ( prevHead >= ( mFifo.size - 1 ) ) ? 0 : prevHead + 1;
237  }
238 
239  Logger::logRawData( packet, FALSE );
240  // checksum is 2's complement of sum of all the data...we calculated checksum
241  // as just the sum, so if we add them together they should be zero
242  if( packet->mChecksum != 0 )
243  {
244  // Invalid checksum; send NAK and keep looking for a packet.
245  if( mTransportLayer )
246  {
247  uint8 payload[2] = { packet->mPacketId, 0 };
248  mTransportLayer->tx
249  (
250  new GarminPacket( ID_NAK_BYTE, payload, sizeof( payload ) ),
251  TRUE
252  );
253  }
254  delete packet;
255  mFifo.head = payloadStart;
256  continue;
257  }
258  else
259  {
260  if( mTransportLayer )
261  {
262  // Send ACK (if required) and push the packet up the stack for processing.
263  // ACK and NAK packets must not be ACKed.
264  if( packet->mPacketId != ID_ACK_BYTE &&
265  packet->mPacketId != ID_NAK_BYTE )
266  {
267  uint8 payload[2] = { packet->mPacketId, 0 };
268  mTransportLayer->tx( new GarminPacket( ID_ACK_BYTE, payload, sizeof( payload ) ), TRUE );
269  }
270  mTransportLayer->rx( packet );
271  }
272  delete packet;
273  }
274  incrementHead( SIZE_OF_FOOTER );
275  return true;
276  }
277  return false;
278 } /* getPacketFromFifo */

372 bool GarminLinkLayer::tx
373  (
374  Packet * aPacket
375  )
376 {
377  GarminPacket * thePacket = dynamic_cast<GarminPacket *>( aPacket );
378 
379  ASSERT( thePacket != NULL );
380 
381  uint8 i;
382  thePacket->mFrameSize = 0;
383 
384  // Frame header (DLE)
385  thePacket->mFrame[ thePacket->mFrameSize++ ] = ID_DLE_BYTE;
386 
387  // Packet ID
388  thePacket->mFrame[ thePacket->mFrameSize++ ] = thePacket->mPacketId;
389 
390  // Payload size, DLE stuffing if necessary
391  thePacket->mFrame[ thePacket->mFrameSize++ ] = thePacket->mPayloadSize;
392  if( thePacket->mPayloadSize == ID_DLE_BYTE )
393  {
394  thePacket->mFrame[ thePacket->mFrameSize++ ] = thePacket->mPayloadSize;
395  }
396 
397  // Payload, DLE stuffing as necessary
398  for( i = 0; i < thePacket->mPayloadSize; i++)
399  {
400  thePacket->mFrame[ thePacket->mFrameSize++ ] = thePacket->mPayload[i];
401  if( thePacket->mPayload[i] == ID_DLE_BYTE )
402  {
403  thePacket->mFrame[ thePacket->mFrameSize++ ] = thePacket->mPayload[i];
404  }
405  }
406 
407  // Checksum, DLE stuffing if necessary
408  thePacket->mFrame[ thePacket->mFrameSize++ ] = thePacket->mChecksum;
409  if( thePacket->mChecksum == ID_DLE_BYTE )
410  {
411  thePacket->mFrame[ thePacket->mFrameSize++ ] = thePacket->mChecksum;
412  }
413 
414  // Frame footer (DLE and ETX bytes)
415  thePacket->mFrame[ thePacket->mFrameSize++ ] = ID_DLE_BYTE;
416  thePacket->mFrame[ thePacket->mFrameSize++ ] = ID_ETX_BYTE;
417 
418  Logger::logRawData( thePacket );
419 
420  return mPhysicalLayer->tx( thePacket->mFrame, thePacket->mFrameSize );
421 }
