119 void CSendSensorDlg::OnBnClickedOk()
120  {
121  UpdateData();
122 
123  switch ( m_action )
124  {
125  case 0:
126  {
127  fmi_sensor_config_request config;
128  memset( &config, 0, sizeof( config ) );
129  config.change_id = GetTickCount();
130  config.unique_id = m_uniqueId;
131  config.display_index = m_sortOrder;
132  config.icon = m_icon.GetCurSel();
133 
134  WideCharToMultiByte( mCom.mClientCodepage, 0, m_name.GetBuffer(), -1, config.name, sizeof( config.name ), NULL, NULL );
135  config.name[sizeof( config.name ) - 1] = '\0';
136 
137  mCom.sendFmiPacket( FMI_SENSOR_CONFIG_REQUEST, (uint8*)&config, sizeof( config ) );
138  break;
139  }
140  case 1:
141  {
142  fmi_sensor_update_request update;
143  memset( &update, 0, sizeof( update ) );
144  update.change_id = GetTickCount();
145  update.unique_id = m_uniqueId;
146  update.severity = m_severity.GetCurSel();
147  update.play_sound = m_playSound;
148  update.record_sensor = m_recordInHistory;
149  WideCharToMultiByte( mCom.mClientCodepage, 0, m_status.GetBuffer(), -1, update.status, sizeof( update.status ), NULL, NULL );
150  update.status[sizeof( update.status ) - 1] = '\0';
151  WideCharToMultiByte( mCom.mClientCodepage, 0, m_description.GetBuffer(), -1, update.description, sizeof( update.description ), NULL, NULL );
152  update.description[sizeof( update.description ) - 1] = '\0';
153 
154  mCom.sendFmiPacket( FMI_SENSOR_UPDATE_REQUEST, (uint8*)&update, sizeof( update ) );
155  break;
156  }
157  case 2:
158  {
159  fmi_sensor_delete_request message;
160  memset( &message, 0, sizeof( message ) );
161  message.change_id = GetTickCount();
162  message.unique_id = m_uniqueId;
163 
164  mCom.sendFmiPacket( FMI_SENSOR_DELETE_REQUEST, (uint8*)&message, sizeof( message ) );
165 
166  break;
167  }
168  case 3:
169  {
170  fmi_sensor_query_display_index_request message;
171  memset( &message, 0, sizeof( message ) );
172  message.change_id = GetTickCount();
173  message.unique_id = m_uniqueId;
174 
175  mCom.sendFmiPacket( FMI_SENSOR_QUERY_DISPLAY_INDEX_REQUEST, (uint8*)&message, sizeof( message ) );
176 
177  break;
178  }
179  }
180 
181  OnOK();
182  }
