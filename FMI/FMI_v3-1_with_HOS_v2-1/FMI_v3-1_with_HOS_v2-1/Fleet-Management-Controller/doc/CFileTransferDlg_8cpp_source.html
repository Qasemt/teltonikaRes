115 void CFileTransferDlg::OnEnChangeFile()
116 {

118

120

125

139  GetDlgItem( IDOK )->EnableWindow( TRUE );

152 void CFileTransferDlg::OnBnClickedFind()
153 {
154  UpdateData();
155 
156  TCHAR workingDirectory[200];
157  // opening a file in another directory changes the current
158  // directory, which will cause problems because the log and data
159  // files are opened relative to the current directory. So, get the
160  // directory now so that it can be restored when the user is done
161  // picking a file.
162  DWORD result = GetCurrentDirectory( 200, workingDirectory );
163  if( result == 0 || result > 200 )
164  {
165  MessageBox( _T("Unable to get current directory"), _T("Severe Error") );
166  OnCancel();
167  return;
168  }
169  CFileDialog dlg
170  (
171  TRUE,
172  file_types[mFileType].extension,
173  NULL,
174  OFN_HIDEREADONLY | OFN_OVERWRITEPROMPT,
175  file_types[mFileType].filter
176  );
177  if( dlg.DoModal() == IDOK )
178  {
179  mFilePath = dlg.GetPathName();
180  UpdateData( FALSE );
181  OnEnChangeFile();
182  }
183  if( !SetCurrentDirectory( workingDirectory ) )
184  {
185  MessageBox( _T("Unable to set current directory"), _T("Severe Error") );
186  OnCancel();
187  return;
188  }
189 } /* OnBnClickedFind() */

200 void CFileTransferDlg::OnBnClickedOk()
201 {
202  UpdateData( TRUE );
203  char file[200];
204 
205  char versionString[35];
206  uint8 version[16];
207  uint8 versionLength;
208 
209  memset( version, 0, sizeof( version ) );
210  WideCharToMultiByte( mCom.mClientCodepage, 0, mVersion, -1, versionString, 34, NULL, NULL );
211  versionString[34] = '\0';
212 
213  if( strncmp( versionString, "0x", 2 ) == 0 )
214  {
215  versionLength = (uint8)UTIL_hex_to_uint8( versionString + 2, version, 16 );
216  }
217  else
218  {
219  versionLength = (uint8)minval( 16, strlen( versionString ) );
220  memmove( version, versionString, versionLength );
221  }
222 
223  WideCharToMultiByte( CP_ACP, 0, mFilePath.GetBuffer(), -1, file, 200, NULL, NULL );
224  file[199] = '\0';
225  fstream open_file( file, ios_base::binary | ios_base::in );
226  if( open_file.good() )
227  {
228  open_file.close();
229 
230  mCom.sendFile( file, versionLength, version, (uint8)mFileType );
231 
232  CFileTransferProgressDlg sending_dlg( this, mCom );
233  sending_dlg.DoModal();
234  }
235  else
236  {
237  mFilePath += _T(" could not be opened ");
238  MessageBox( mFilePath, _T("Error!") );
239  }
240  open_file.close();
241  OnOK();
242 } /* OnBnClickedOk */
